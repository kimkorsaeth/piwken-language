\newcounter{defcount} % enumerate definitions
\newcounter{dictwordcount} % total word count

\newcommand{\dictletter}[1]{
    \section*{\centering #1}
}

\newcommand{\dictentry}[3]{
    \refstepcounter{dictwordcount}
    \setcounter{defcount}{0}
    \noindent\textbf{\large #1} {/#2/} \textit{#3.}
}

\newcommand{\dictappend}[2]{
    % \refstepcounter{dictwordcount} % uncomment this if these should count as unique words
    \noindent\textbf{#1} {/#2/}
}

\newcommand{\dictdef}[1]{
    \refstepcounter{defcount}
    \textbf{\thedefcount}\penalty10000 { #1}
}

\WithSuffix\newcommand\dictdef*[1]{%
    { #1}
}
